async function salvarServico() { const titulo = document.getElementById("titulo").value; const descricao = document.getElementById("descricao").value; const valor = document.getElementById("valor").value; if (!titulo || !valor) return alert("Preencha os campos obrigatórios!"); const novoServico = { id: Date.now(), titulo, descricao, valor: parseFloat(valor.replace(',', '.')), // Converte para número status: "disponivel" }; const url = "https://api.github.com/repos/jfbj54-arch/rs-connect/contents/servicos.json" try { // Busca o arquivo atual const resposta = await fetch(url, { headers: { "Authorization": token } }); if (!resposta.ok) throw new Error("Erro ao buscar arquivo"); const dados = await resposta.json(); // Decodifica UTF-8 corretamente (melhor que atob para caracteres especiais) const conteudoAtual = JSON.parse(decodeURIComponent(escape(window.atob(dados.content)))); if (!conteudoAtual.servicos) conteudoAtual.servicos = []; conteudoAtual.servicos.push(novoServico); // Codifica para Base64 const novoConteudo = window.btoa(unescape(encodeURIComponent(JSON.stringify(conteudoAtual, null, 2)))); // Faz o Update const update = await fetch(url, { method: "PUT", headers: { "Content-Type": "application/json", "Authorization": token }, body: JSON.stringify({ message: `Adicionado serviço: ${titulo}`, content: novoConteudo, sha: dados.sha }) }); if (update.ok) { alert("Serviço cadastrado com sucesso!"); document.querySelectorAll("input").forEach(i => i.value = ""); // Limpa campos } } catch (erro) { console.error(erro); alert("Erro ao salvar. Verifique o console."); } }
