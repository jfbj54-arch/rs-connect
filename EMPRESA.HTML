async function salvarServico() { const titulo = document.getElementById("titulo").value; const descricao = document.getElementById("descricao").value; const valor = document.getElementById("valor").value; const novoServico = { id: Date.now(), titulo, descricao, valor, status: "disponivel" }; const url = "https://api.github.com/repos/jfbj54-arch/rs-connect/contents/servicos.json"; const token = "Bearer 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"; const resposta = await fetch(url, { headers: { "Authorization": token } }); const dados = await resposta.json(); const conteudoAtual = JSON.parse(atob(dados.content)); if (!conteudoAtual.servicos) { conteudoAtual.servicos = []; } conteudoAtual.servicos.push(novoServico); const novoConteudo = btoa(JSON.stringify(conteudoAtual, null, 2)); await fetch(url, { method: "PUT", headers: { "Content-Type": "application/json", "Authorization": token }, body: JSON.stringify({ message: "Novo serviço adicionado", content: novoConteudo, sha: dados.sha }) }); alert("Serviço cadastrado com sucesso!"); }
